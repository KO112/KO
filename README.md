# Calculate the stars for p-values
pvalue_stars <- function(p_values) {
  case_when(
    p_values <= 0.001 ~ "***",
    p_values <= 0.01 ~ "**",
    p_values <= 0.05 ~ "*",
    p_values <= 0.1 ~ ".",
    TRUE ~ "---"
  ) %>%
    KO::pad_vector(vec = ., alignment = "R") %>%
    return()
}
